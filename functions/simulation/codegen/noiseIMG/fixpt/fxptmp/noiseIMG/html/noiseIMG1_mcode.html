1%% noiseIMG() function
  2%   Creates simulated noisy image
  3%
  4%
  5%% INPUT options:
  6% * _simg:_ Image dimensions (Y,X)
  7% * _nump:_ Number of simulated particles
  8% * _sPSF:_ Radius of PSF in pixels
  9% * _ftype:_ Type of filter for blur, options are: ['none', 'gaussian', 'bessel', 'poisson', 'laplacian', 'disk']
 10% * _ntype:_ Type of noise, options are: ['none', 'gaussian', 'poisson', 'laplacian', 'disk']
 11% * _sr:_ Desired final signal to noise ratio (sig vs background
 12%
 13%% OUTPUT arguments:
 14% * _nimg:_ Noisy particle image simulation
 15% * _y:_    y-coordinates of particle seeds, mainly for debugging
 16% * _x:_    x-coordinates of particle seeds, mainly for debugging
 17%
 18
 19
 20function [nimg, y ,x ] = noiseIMG(simg, nump, sPSF, ftype, ntype, sr)
 21coder.extrinsic('besselj','imnoise'); % besselj is not impl. in C
 22% set default arguments:
 23if isempty(ftype)
 24    ftype = 'gaussian';
 25end
 26
 27if strcmp(ftype, 'bessel')
 28    % apply PSF  with 1st order bessel function
 29    NA = 1.2;       % numerical aperture of objective
 30    nw = 1.33;      % refractive index of second medium, 1.33 for water
 31    lambda = 532/1000;   % excitation wavelendth used in nm
 32    np = 51;
 33    c = repmat(-np:np,2*np+1,1);  % Array of column numbers, relative to center
 34    r = repmat((-np:np)',1,2*np+1);  % Array of row numbers, relative to center
 35    d = sqrt(r.*r + c.*c);  % matrix of distances from the center, in pixels
 36    
 37    rpos = d/(np*2+1);  % matrix of distances, microns * grid size -> nm per pixel
 38    v = (2*pi/lambda)*(NA/nw)*rpos;  % array of 'v' values
 39    besv = besselj(1,v)./v;
 40    psf = 4.0*(besv.*conj(besv));
 41    psf(np+1,np+1) = 1.0;  % correct central value
 42    psf = psf / max(psf(:)); % normalize
 43    
 44end
 45
 46if isempty(sPSF)
 47    sPSF = 5;
 48end
 49
 50
 51
 52% if Input is another image...
 53if max(size(simg)) > 3
 54    nimg = simg;
 55    simg = size(simg);
 56    bimg = double(zeros(simg));
 57else
 58    nimg = double(zeros(simg)); % allocate memory for images
 59    bimg = nimg;
 60    
 61    % calculate and add randomized pixels
 62    for i=1:nump
 63        randXY = floor(rand(size(simg)) .* (simg - 1)) + 1;
 64        nimg(randXY(1),randXY(2)) = 1;
 65        [y,x] = deal(randXY(1,2),randXY(1,1));
 66    end
 67    
 68    % perform gaussian convolution (gaussian blur)
 69    switch ftype
 70        
 71        case 'none'
 72        case 'bessel'
 73        otherwise
 74            IF = fspecial(ftype,[6,6]*sPSF,sPSF); % create filter with size = 6*PSF
 75            nimg = imfilter(nimg,IF,'same'); % create and apply filter
 76            
 77            % normalize max to 1
 78            nimg = nimg ./ max(nimg(:));
 79            
 80    end
 81    
 82end
 83
 84
 85
 86
 87
 88
 89% add noise
 90switch ntype
 91    case 'poisson'
 92        nimg = imnoise(nimg * 1E-11, ntype);
 93        % normalize again to 1
 94        nimg = nimg ./ max(nimg(:));
 95    case 'none'
 96        
 97    otherwise
 98        nimg = imnoise(nimg, ntype);
 99end
100
101% add background noise floor
102
103if ~isempty(sr)
104    bimg = imnoise(ones(simg), 'gaussian',0.2,1);
105    nimg = nimg + bimg / sr;
106    nimg = nimg ./ max(nimg(:)); %normalize
107end
108
109% show result -> for debugging;
110%imtool(nimg,[0,1]);
111end
